date_selected <- as.Date(c("0000-01-01", "0000-04-01",
                                            "0000-07-01",
                                            "0000-10-01",
                                            "0000-12-31"
                                            ))
df <- filter(ncdc_temp, date %in% date_selected)
p <- ggplot(data = ncdc_temp,
            mapping = 
              aes( x = date,
                   y = temperature ) )

p + geom_line( aes( color = location ),show.legend = F) +
  geom_point(data = df) +
  labs(x = "month") +
  scale_x_date(breaks = date_selected,
               date_labels = "%b")
